## 模範回答
def hanoi(n, source ='A', auxiliary ='B', target ='C'):
    if n == 1:
        print(f"円盤 {n}を {source}から {target}へ移動")
        return
    hanoi(n - 1, source, target, auxiliary)
    print(f"円盤 {n}を {source}から {target}へ移動")
    hanoi(n - 1, auxiliary, source, target)

n = int(input())
hanoi(n, 'A', 'B', 'C')
